def is_prime(n: int):
    if n < 2:
        return False
    elif n == 2:
        return True
    elif n % 2 == 0:
        return False
    for d in range(3, n, 2):
        if n % d == 0:
            return False
        if n < d**2:
            break
    return True

def int2base(n: int, b: int):
    assert b <= len(DIGITS), f"Base cannot exceed {len(DIGITS)}."
    assert not n < 0 and isinstance(n, int), "n should be a positive integer."
    if n == 0:
        return DIGITS[0]

    res = ""
    while n != 0:
        n, i = divmod(n, b)
        res = DIGITS[i] + res
    return res

def base2int(n: str, b: int):
    assert b <= len(DIGITS), f"Base cannot exceed {len(DIGITS)}."
    res = 0
    for d in n:
        d_num = DIGITS.index(d)
        assert d_num < b, f"{n} is not a valid number in base {b}."
        res = b * res + d_num
    return res

def natural_numbers(start: int = 0):
    n = start
    while True:
        yield n
        n += 1

def even_numbers(start: int = 0):
    NN = natural_numbers(start)
    for n in NN:
        yield 2 * n

def prime_numbers():
    NN = natural_numbers(2)
    for n in NN:
        if is_prime(n):
            yield n

n_first_primes = 100

bit_size = 2

primes_dict = {}
bit_size_primes = []

for _, p in zip(range(n_first_primes), prime_numbers()):
    curr_bit_size = len_in_bit(p)
    if bit_size < curr_bit_size:
        primes_dict[bit_size] = bit_size_primes
        bit_size_primes = []
        bit_size = curr_bit_size
    bit_size_primes.append(p)

for bit_size, ps in primes_dict.items():
    print(f"{bit_size} bits\t:\t{ps}")
